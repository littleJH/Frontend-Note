### 手写生成器函数

关键实现：

- `next()` 方法
- 返回 `{done, value}`

```jsx
function myGenerator(list) {
	// 借助闭包记录当前索引
  let index = 0
  return {
    next: function() {
      const done =  index >= list.length - 1
      const value = !done ? list[index] : undefined
      index++
      return {
        done,
        value
      }
    }
  }
}

const iterator = myGenerator([1, 2, 3])
console.log("🚀 ~ iterator.next():", iterator.next())
console.log("🚀 ~ iterator.next():", iterator.next())
console.log("🚀 ~ iterator.next():", iterator.next())
console.log("🚀 ~ iterator.next():", iterator.next())
```
